## HEALTH CHECK
GET /health

## MEMBER ENDPOINTS
POST /api/v1/members                             # Create a new member 
# Create a new member
curl -X POST http://localhost:8001/api/v1/members \
  -H "Content-Type: application/json" \
  -d '{
    "firstName": "John",
    "lastName": "Doe",
    "email": "john.doe@example.com",
    "phone": "555-123-4567",
    "address": "123 Main St",
    "dateOfBirth": "1990-01-15T00:00:00Z",
    "emergencyContactName": "Jane Doe",
    "emergencyContactPhone": "555-987-6543"
  }'

GET /api/v1/members                              # List all members 
# List all members
curl -X GET http://localhost:8001/api/v1/members

# List members with pagination
curl -X GET "http://localhost:8001/api/v1/members?page=1&size=10"

GET /api/v1/members/{id}                         # Get a specific member
# Get a specific member
curl -X GET http://localhost:8001/api/v1/members/1

PUT /api/v1/members/{id}                         # Update a member 
# Update a member
curl -X PUT http://localhost:8001/api/v1/members/1 \
  -H "Content-Type: application/json" \
  -d '{
    "firstName": "John",
    "lastName": "Smith",
    "email": "john.smith@example.com",
    "phone": "555-123-8888",
    "address": "456 Oak Ave"
  }'

DELETE /api/v1/members/{id}                      # Delete a member
# Delete a member
curl -X DELETE http://localhost:8001/api/v1/members/1

GET /api/v1/members/{memberID}/memberships       # Get all memberships for a member
# Get all memberships for a member
curl -X GET http://localhost:8001/api/v1/members/1/memberships

GET /api/v1/members/{memberID}/active-membership # Get active membership for a member
# Get active membership for a member
curl -X GET http://localhost:8001/api/v1/members/1/active-membership

GET /api/v1/members/{memberID}/assessments       # Get all assessments for a member
# Get all assessments for a member
curl -X GET http://localhost:8001/api/v1/members/1/assessments

## MEMBERSHIP ENDPOINTS
POST /api/v1/memberships                        # Create a new membership

  -H "Content-Type: application/json" \
  -d '{
    "membershipName": "Premium",
    "description": "Full access to gym facilities and group classes",
    "duration": 3,
    "price": 49.99,
    "isActive": true
  }'

curl -X GET http://localhost:8001/api/v1/memberships

# Get only active memberships

GET /api/v1/memberships/{id}                    # Get a specific membership

DELETE /api/v1/memberships/{id}                 # Delete a membership
