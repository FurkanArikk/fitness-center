# CLASS SERVICE API ENDPOINTS

## HEALTH CHECK
GET /health
# Check service health
curl -X GET http://localhost:8005/health

## CLASS ENDPOINTS
POST /api/v1/classes                             # Create a new class 
# Create a new class
curl -X POST http://localhost:8005/api/v1/classes \
  -H "Content-Type: application/json" \
  -d '{
    "class_name": "Yoga Flow",
    "description": "A flowing sequence of poses that synchronizes breath with movement",
    "duration": 60,
    "capacity": 20,
    "difficulty": "Beginner",
    "is_active": true
  }'

GET /api/v1/classes                              # List all classes 
# List all classes
curl -X GET http://localhost:8005/api/v1/classes

# List only active classes
curl -X GET "http://localhost:8005/api/v1/classes?active=true"

GET /api/v1/classes/{id}                         # Get a specific class
# Get a specific class
curl -X GET http://localhost:8005/api/v1/classes/1

PUT /api/v1/classes/{id}                         # Update a class 
# Update a class
curl -X PUT http://localhost:8005/api/v1/classes/1 \
  -H "Content-Type: application/json" \
  -d '{
    "class_name": "Power Yoga",
    "description": "A vigorous fitness-based approach to Vinyasa-style yoga",
    "duration": 75,
    "capacity": 15,
    "difficulty": "Intermediate",
    "is_active": true
  }'

DELETE /api/v1/classes/{id}                      # Delete a class
# Delete a class
curl -X DELETE http://localhost:8005/api/v1/classes/1

## SCHEDULE ENDPOINTS
POST /api/v1/schedules                          # Create a new schedule
# Create a new schedule
curl -X POST http://localhost:8005/api/v1/schedules \
  -H "Content-Type: application/json" \
  -d '{
    "class_id": 1,
    "trainer_id": 5,
    "room_id": 2,
    "start_time": "09:00:00",
    "end_time": "10:00:00",
    "day_of_week": "Monday",
    "status": "active"
  }'

GET /api/v1/schedules                           # Get all schedules (optional ?status=active filter)
# Get all schedules
curl -X GET http://localhost:8005/api/v1/schedules

# Get only active schedules
curl -X GET "http://localhost:8005/api/v1/schedules?status=active"

GET /api/v1/schedules/{id}                      # Get a specific schedule
# Get a specific schedule
curl -X GET http://localhost:8005/api/v1/schedules/1

GET /api/v1/schedules/class/{classId}           # Get schedules for a specific class
# Get schedules for a specific class
curl -X GET http://localhost:8005/api/v1/schedules/class/1

PUT /api/v1/schedules/{id}                      # Update a schedule

curl -X PUT http://localhost:8005/api/v1/schedules/1 \
  -H "Content-Type: application/json" \
  -d '{
    "class_id": 1,
    "trainer_id": 5,
    "room_id": 3,
    "start_time": "10:00:00",
    "end_time": "11:00:00",
    "day_of_week": "Monday",
    "status": "active"
  }'

DELETE /api/v1/schedules/{id}                   # Delete a schedule

curl -X DELETE http://localhost:8005/api/v1/schedules/1

## BOOKING ENDPOINTS
POST /api/v1/bookings                          # Create a new booking
# Create a new booking
curl -X POST http://localhost:8005/api/v1/bookings \
  -H "Content-Type: application/json" \
  -d '{
    "schedule_id": 1,
    "member_id": 3,
    "booking_date": "2023-07-24T08:00:00Z"
  }'

GET /api/v1/bookings                           # Get all bookings (optional filters)
# Get all bookings
curl -X GET http://localhost:8005/api/v1/bookings

# Get bookings with filters
curl -X GET "http://localhost:8005/api/v1/bookings?status=attended&date=2023-07-15"

GET /api/v1/bookings/{id}                      # Get a specific booking
# Get a specific booking
curl -X GET http://localhost:8005/api/v1/bookings/1

GET /api/v1/bookings/member/{memberId}         # Get bookings for a specific member
# Get bookings for a specific member
curl -X GET http://localhost:8005/api/v1/bookings/member/3

PUT /api/v1/bookings/{id}/status               # Update booking status
# Update booking status
curl -X PUT http://localhost:8005/api/v1/bookings/1/status \
  -H "Content-Type: application/json" \
  -d '{
    "attendance_status": "attended"
  }'

POST /api/v1/bookings/{id}/feedback            # Add feedback to a booking
# Add feedback to a booking
curl -X POST http://localhost:8005/api/v1/bookings/1/feedback \
  -H "Content-Type: application/json" \
  -d '{
    "rating": 5,
    "comment": "Great class, very energizing!"
  }'

DELETE /api/v1/bookings/{id}                   # Cancel a booking
# Cancel a booking
curl -X DELETE http://localhost:8005/api/v1/bookings/1
